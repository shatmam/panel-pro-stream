// BUSCA ESTA FUNCIÓN DENTRO DE TU INDEX.HTML Y REEMPLÁZALA: async function load(quiet = false) { if(!quiet) setStatus("Cargando datos..."); hideErr(); try { const data = await api("/api/dashboard"); allRows = data.rows; // ACTUALIZAR TUS CONTADORES (Usando los IDs de tu diseño original) if($("statTotal")) $("statTotal").textContent = data.stats.total; if($("statActive")) $("statActive").textContent = data.stats.active; if($("statExpired")) $("statExpired").textContent = data.stats.expired; if($("statAvailable")) $("statAvailable").textContent = data.stats.available; render(); if(!quiet) setStatus("Listo"); } catch(e) { showErr(e.message); setStatus("Error al cargar"); // Si falla el login, pedimos la clave de nuevo const pass = prompt("Admin Key:"); if(pass){ localStorage.setItem("admin_key", pass); load(); } } } // BUSCA ESTA PARTE EN TU RENDER PARA QUE NO TE BORRE LOS BOTONES: function render() { const q = $("q").value.toLowerCase(); const f = $("f").value; const body = $("tableBody"); body.innerHTML = ""; const filtered = allRows.filter(r => { const matchQ = !q || (r.nombre + r.correo + r.servicio).toLowerCase().includes(q); const matchF = !f || r.estado === f; return matchQ && matchF; }); filtered.forEach(r => { const tr = document.createElement("tr"); // Aquí mantenemos tu lógica de badges y botones originales... const isVencido = r.estado === "VENCIDO"; const isDisponible = r.estado === "DISPONIBLE"; tr.innerHTML = `
${r.servicio}
${r.correo}
${r.nombre} ${r.estado} ${r.dias} ${isVencido ? `` : ''} ${isDisponible ? `` : ''} `; body.appendChild(tr); }); }
